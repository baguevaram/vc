1linkprecision mediump float;

2linkuniform sampler2D texture;

3linkuniform float mask[9];

4linkuniform vec2 offset;

5link// interpolated color (same name as in vertex shader)

6linkvarying vec4 vVertexColor;

7link// interpolated texcoord (same name as in vertex shader)

8linkvarying vec2 vTexCoord;

9linkvoid main() {

10link    vec2 tex0 = vTexCoord.st + vec2(-offset.s, -offset.t);

11link    vec2 tex1 = vTexCoord.st + vec2(0.0, -offset.t);

12link    vec2 tex2 = vTexCoord.st + vec2(offset.s, -offset.t);

13link    vec2 tex3 = vTexCoord.st + vec2(-offset.s, 0.0);

14link    vec2 tex4 = vTexCoord.st + vec2(0.0, 0.0);

15link    vec2 tex5 = vTexCoord.st + vec2(offset.s, 0.0);

16link    vec2 tex6 = vTexCoord.st + vec2(-offset.s, offset.t);

17link    vec2 tex7 = vTexCoord.st + vec2(0.0, offset.t);

18link    vec2 tex8 = vTexCoord.st + vec2(offset.s, offset.t);

19link    vec4 rgba[9];

20link    rgba[0] = texture2D(texture, tex0);

21link    rgba[1] = texture2D(texture, tex1);

22link    rgba[2] = texture2D(texture, tex2);

23link    rgba[3] = texture2D(texture, tex3);

24link    rgba[4] = texture2D(texture, tex4);

25link    rgba[5] = texture2D(texture, tex5);

26link    rgba[6] = texture2D(texture, tex6);

27link    rgba[7] = texture2D(texture, tex7);

28link    rgba[8] = texture2D(texture, tex8);

29link    vec4 convolution;

30link    for (int i=0; i<9; i++){

31link        convolution+=rgba[i]*mask[i];

32link    }

33link    gl_FragColor = vec4(convolution.rgb, 1.0) * vVertexColor;

34link}
